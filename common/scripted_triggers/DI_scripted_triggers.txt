    OR = {
        scope:track = flag:Wanderer
        scope:track = flag:Seasoned
        scope:track = flag:Foot
        scope:track = flag:Bow
        scope:track = flag:Horse
        scope:track = flag:Wit
        scope:track = flag:Venator
        scope:track = flag:Falconer
        scope:track = flag:Bandit
        scope:track = flag:Trickster
        scope:track = flag:Thief
        scope:track = flag:Poacher
        scope:track = flag:Marauder
    }
}

DI_char_spawner_gender_check = {
    OR = {
        AND = {
            is_female = yes
            scope:DI_story = { has_variable = DI_char_spawner_female }
        }
        AND = {
            is_female = no
            NOT = {scope:DI_story = { has_variable = DI_char_spawner_female }}
        }
    }
}

DI_char_has_parent_ruler = {
    OR = {
        trigger_if = { ## Has father
            limit = {
                has_father = yes
            }
            father = { is_ruler = yes }
        }
        trigger_if = { ## Has mother
            limit = {
                has_mother = yes
            }
            mother = { is_ruler = yes }
        }
    }
}
